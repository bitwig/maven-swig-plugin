291     public void execute()
292         throws MojoExecutionException, MojoFailureException
293     {
294         if ( skip )
295         {
296             getLog().warn( "SKIPPED Running SWIG compiler on " + source + " ..." );
297             return;
298         }
299 
300         if ( !sourceDirectory.endsWith( "/" ) )
301         {
302             sourceDirectory = sourceDirectory + "/";
303         }
304         File sourceDir = new File( sourceDirectory );
305         File sourceFile = new File( sourceDir, source );
306         if ( !sourceDir.exists() || !sourceFile.exists() )
307         {
308             getLog().info( "No SWIG sources found" );
309             return;
310         }
311 
312         os = NarUtil.getOS( os );
313 
314         Linker linker = new Linker(NarUtil.getLinkerName( architecture, os, null ));
315         narManager = new NarManager( getLog(), localRepository, project, architecture, os, linker );
316 
317         targetDirectory = new File( targetDirectory, cpp ? "c++" : "c" );
318         if ( !targetDirectory.exists() )
319         {
320             targetDirectory.mkdirs();
321         }
322 
323         if ( project != null )
324         {
325             project.addCompileSourceRoot( javaTargetDirectory );
326             project.addCompileSourceRoot( targetDirectory.getPath() );
327         }
328 
329         if ( packageName != null )
330         {
331             if ( !javaTargetDirectory.endsWith( "/" ) )
332             {
333                 javaTargetDirectory = javaTargetDirectory + "/";
334             }
335             javaTargetDirectory += packageName.replace( '.', File.separatorChar );
336         }
337 
338         if ( !FileUtils.fileExists( javaTargetDirectory ) )
339         {
340             FileUtils.mkdir( javaTargetDirectory );
341         }
342 
343         // make sure all NAR dependencies are downloaded and unpacked
344         // even if packaging is NOT nar
345         // in nar packaging, downloading happens in generate-sources phase and
346         // thus may be too late
347         // unpacking happens in process-sources which is definitely too late
348         // so we need to handle this here ourselves.
349         List narArtifacts = narManager.getNarDependencies( "compile" );
350         narManager.downloadAttachedNars( narArtifacts, remoteArtifactRepositories, artifactResolver, null );
351         narManager.unpackAttachedNars( narArtifacts, archiverManager, null, os );
352 
353         File swig, swigInclude, swigJavaInclude;
354         if ( exec == null )
355         {
356             // NOTE, since a project will just load this as a plugin, there is
357             // no way to look up the org.swig:nar-swig dependency, so we hardcode
358             // that in here, but it is configurable in the configuration part of this plugin.
359             Artifact swigJar =
360                 new DefaultArtifact( groupId, artifactId, VersionRange.createFromVersion( version ), "compile", "jar",
361                                      "", artifactHandler );
362 
363             // download jar file
364             try
365             {
366                 getLog().debug( "maven-swig-plugin: Resolving " + swigJar );
367                 artifactResolver.resolve( swigJar, remoteArtifactRepositories, localRepository );
368             }
369             catch ( ArtifactNotFoundException e )
370             {
371                 String message = "Jar not found " + swigJar;
372                 throw new MojoExecutionException( message, e );
373             }
374             catch ( ArtifactResolutionException e )
375             {
376                 String message = "Jar cannot be resolved " + swigJar;
377                 throw new MojoExecutionException( message, e );
378             }
379 
380             NarInfo info = narManager.getNarInfo( swigJar );
381             getLog().debug( info.toString() );
382             NarArtifact swigNar = new NarArtifact( swigJar, info );
383 
384             // download attached nars, in which the executable and the include files sit
385             List swigNarArtifacts = new ArrayList();
386             swigNarArtifacts.add( swigNar );
387             narManager.downloadAttachedNars( swigNarArtifacts, remoteArtifactRepositories, artifactResolver, null );
388             narManager.unpackAttachedNars( swigNarArtifacts, archiverManager, null, os );
389 
390             swig = new File( narManager.getNarFile( swigNar ).getParentFile(), "nar" );
391             swigInclude = new File( swig, "include" );
392             swigJavaInclude = new File( swigInclude, "java" );
393             swig = new File( swig, "bin" );
394             swig = new File( swig, NarUtil.getAOL( architecture, os, linker, null ).toString() );
395             swig = new File( swig, "swig" );
396         }
397         else
398         {
399             swig = new File( exec );
400             swigInclude = null;
401             swigJavaInclude = null;
402         }
403 
404         File targetFile = targetDirectory;
405         SourceInclusionScanner scanner =
406             new StaleSourceScanner( staleMillis, Collections.singleton( source ), Collections.EMPTY_SET );
407         String extension = "." + FileUtils.getExtension( source );
408         SuffixMapping mapping = new SuffixMapping( extension, extension + ".flag" );
409         scanner.addSourceMapping( mapping );
410         try
411         {
412             Set files = scanner.getIncludedSources( sourceDir, targetFile );
413 
414             if ( !files.isEmpty() || force )
415             {
416                 if ( cleanOutputDirectory && ( packageName != null ) && FileUtils.fileExists( javaTargetDirectory ) )
417                 {
418                     getLog().info( "Cleaning " + javaTargetDirectory );
419                     String[] filesToRemove =
420                         FileUtils.getFilesFromExtension( javaTargetDirectory, new String[] { "java" } );
421                     for ( int i = 0; i < filesToRemove.length; i++ )
422                     {
423                         File f = new File( filesToRemove[i] );
424                         f.delete();
425                     }
426                 }
427 
428                 getLog().info( ( force ? "FORCE " : "" ) + "Running SWIG compiler on " + source + " ..." );
429                 int error = runCommand( generateCommandLine( swig, swigInclude, swigJavaInclude ) );
430                 if ( error != 0 )
431                 {
432                     throw new MojoFailureException( "SWIG returned error code " + error );
433                 }
434                 File flagFile = new File( targetDirectory, source + ".flag" );
435                 FileUtils.fileDelete( flagFile.getPath() );
436                 FileUtils.fileWrite( flagFile.getPath(), "" );
437             }
438             else
439             {
440                 getLog().info( "Nothing to swig - all classes are up to date" );
441             }
442         }
443         catch ( InclusionScanException e )
444         {
445             throw new MojoExecutionException( "IDLJ: Source scanning failed", e );
446         }
447         catch ( IOException e )
448         {
449             throw new MojoExecutionException( "SWIG: Creation of timestamp flag file failed", e );
450         }
451     }

453     private String[] generateCommandLine( File swig, File swigInclude, File swigJavaInclude )
454         throws MojoExecutionException, MojoFailureException
455     {
456 
457         List cmdLine = new ArrayList();
458 
459         cmdLine.add( swig.toString() );
460 
461         if ( fakeVersion != null )
462         {
463             cmdLine.add( "-fakeversion" );
464             cmdLine.add( fakeVersion );
465         }
466 
467         if ( getLog().isDebugEnabled() )
468         {
469             cmdLine.add( "-v" );
470         }
471 
472         // FIXME hardcoded
473         cmdLine.add( "-java" );
474 
475         if ( cpp )
476         {
477             cmdLine.add( "-c++" );
478         }
479 
480         // defines
481         if ( defines != null )
482         {
483             for ( Iterator i = defines.iterator(); i.hasNext(); )
484             {
485                 cmdLine.add( "-D" + i.next() );
486             }
487         }
488 
489         // warnings
490         if ( noWarn != null )
491         {
492             String noWarns[] = noWarn.split( ",| " );
493             for ( int i = 0; i < noWarns.length; i++ )
494             {
495                 cmdLine.add( "-w" + noWarns[i] );
496             }
497         }
498 
499         if ( warnAll )
500         {
501             cmdLine.add( "-Wall" );
502         }
503 
504         if ( warnError )
505         {
506             cmdLine.add( "-Werror" );
507         }
508 
509         // output file
510         String baseName = FileUtils.basename( source );
511         cmdLine.add( "-o" );
512         File outputFile =
513             outFile != null ? new File( targetDirectory, outFile ) : new File( targetDirectory, baseName
514                 + ( cpp ? "cxx" : "c" ) );
515         cmdLine.add( outputFile.toString() );
516 
517         // package for java code
518         if ( packageName != null )
519         {
520             cmdLine.add( "-package" );
521             cmdLine.add( packageName );
522         }
523 
524         // output dir for java code
525         cmdLine.add( "-outdir" );
526         cmdLine.add( javaTargetDirectory );
527 
528         // user added include dirs
529         if ( includePaths != null )
530         {
531             for ( Iterator i = includePaths.iterator(); i.hasNext(); )
532             {
533                 cmdLine.add( "-I" + i.next() );
534             }
535         }
536         // default include dirs
537         cmdLine.add( "-I" + "src/main/include" );
538         cmdLine.add( "-I" + sourceDirectory );
539 
540         // NAR dependency include dirs
541         List narIncludes = narManager.getNarDependencies( "compile" );
542         for ( Iterator i = narIncludes.iterator(); i.hasNext(); )
543         {
544             Artifact narInclude = (Artifact) i.next();
545             File narIncludeDir = new File( narManager.getNarFile( narInclude ).getParentFile(), "nar" );
546             narIncludeDir = new File( narIncludeDir, "include" );
547             if ( narIncludeDir.isDirectory() )
548             {
549                 cmdLine.add( "-I" + narIncludeDir );
550             }
551         }
552 
553         // system swig include dirs
554         if ( swigJavaInclude != null )
555             cmdLine.add( "-I" + swigJavaInclude.toString() );
556         if ( swigInclude != null )
557             cmdLine.add( "-I" + swigInclude.toString() );
558 
559         // swig file
560         cmdLine.add( sourceDirectory + source );
561 
562         StringBuffer b = new StringBuffer();
563         for ( Iterator i = cmdLine.iterator(); i.hasNext(); )
564         {
565             b.append( (String) i.next() );
566             if ( i.hasNext() )
567                 b.append( ' ' );
568         }
569         getLog().info( b.toString() );
570 
571         return (String[]) cmdLine.toArray( new String[cmdLine.size()] );
572     }

574     private int runCommand( String[] cmdLine )
575         throws MojoExecutionException
576     {
577         try
578         {
579             Runtime runtime = Runtime.getRuntime();
580             Process process = runtime.exec( cmdLine );
581             StreamGobbler errorGobbler = new StreamGobbler( process.getErrorStream(), true );
582             StreamGobbler outputGobbler = new StreamGobbler( process.getInputStream(), false );
583 
584             errorGobbler.start();
585             outputGobbler.start();
586             return process.waitFor();
587         }
588         catch ( Throwable e )
589         {
590             throw new MojoExecutionException( "Could not launch " + cmdLine[0], e );
591         }
592     }

594     class StreamGobbler
595         extends Thread
596     {
597         InputStream is;
598 
599         boolean error;
600 
601         StreamGobbler( InputStream is, boolean error )
602         {
603             this.is = is;
604             this.error = error;
605         }
606 
607         public void run()
608         {
609             try
610             {
611                 BufferedReader reader = new BufferedReader( new InputStreamReader( is ) );
612                 String line = null;
613                 while ( ( line = reader.readLine() ) != null )
614                 {
615                     if ( error )
616                     {
617                         getLog().error( line );
618                     }
619                     else
620                     {
621                         getLog().debug( line );
622                     }
623                 }
624                 reader.close();
625             }
626             catch ( IOException e )
627             {
628                 e.printStackTrace();
629             }
630         }
631     }
